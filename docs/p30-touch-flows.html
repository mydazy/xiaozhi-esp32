6 配置参数说明

v5.0 推荐 config struct · 一次性传入 · 运行时不变。

/* RF 抗扰模式 · 按板级 i2c 干扰程度选档 */
typedef enum {
    AXS5106L_RF_NORMAL = 0,   /* WiFi 板 / 无 4G 干扰 · 较宽容 */
    AXS5106L_RF_STRICT = 1,   /* 4G 共线 · storm/debounce/guard 加严 */
} axs5106l_rf_mode_t;

/* 5 个事件回调类型（一事件一类型） */
typedef void (*axs5106l_wake_cb_t)       (void *ctx);
typedef void (*axs5106l_click_cb_t)      (int16_t x, int16_t y, void *ctx);
typedef void (*axs5106l_swipe_cb_t)      (int16_t dx, int16_t dy, void *ctx);
typedef void (*axs5106l_long_press_cb_t) (bool is_release, int16_t x, int16_t y, void *ctx);

typedef struct {
    /* 硬件 · 必填 */
    i2c_worker_handle_t   worker;
    gpio_num_t            rst_gpio, int_gpio;
    uint16_t              width, height;

    /* RF 抗扰 · 按板级 i2c 干扰单独配置 */
    axs5106l_rf_mode_t    rf_mode;     /* 默认 NORMAL · 4G 板传 STRICT */

    /* 回调 · 5 个独立 cb · 不需要的传 NULL */
    void                 *cb_ctx;
    axs5106l_wake_cb_t        on_wake;
    axs5106l_click_cb_t       on_click;
    axs5106l_click_cb_t       on_double_click;
    axs5106l_swipe_cb_t       on_swipe;
    axs5106l_long_press_cb_t  on_long_press;
} axs5106l_touch_config_t;

8 寄存器说明

寄存器地址方向作用
REG_DATA0x01R · 6 字节buf[0] 硬件手势（不用）· buf[1] touch_num · X=(b[2]&0x0F)<<8|b[3] · Y=(b[4]&0x0F)<<8|b[5]
REG_FW_VER0x05R · 2固件版本 V2907=0x2907
REG_CHIP_ID0x08R · 3全 0/0xFF 视为故障
REG_SLEEP0x19W · 1写 0x03 进 sleep
REG_RESET0xF0W · 5软复位 {B3,55,AA,34,01}
  • X 死区软件补偿：raw [9..272] → 屏 [0..283] · x = (raw - 9) × 284 / 263
  • ISR（IRAM）：仅 int_edge_count++ · 不读 I²C · < 1μs
  • LVGL task：30ms 周期读 REG_DATA + 手势状态机 + 三层 RF 过滤
  • storm 检测：1s rolling 窗采样 int_edge_count · 超阈（NORMAL 20/STRICT 12）→ mute

9 典型配置示例

9.1 P30-4G · 4G 共线 → RF_STRICT

axs5106l_touch_handle_t touch_ = nullptr;

void InitializeTouchHardware() {
    axs5106l_touch_config_t cfg = {
        .worker          = i2c_worker_,
        .rst_gpio        = TOUCH_RST_NUM,         /* GPIO_4 */
        .int_gpio        = TOUCH_INT_NUM,         /* GPIO_5 */
        .width           = DISPLAY_WIDTH,         /* 284 */
        .height          = DISPLAY_HEIGHT,        /* 240 */
        .rf_mode         = AXS5106L_RF_STRICT,    /* 4G 共线加严 */
        .cb_ctx          = this,
        .on_wake         = &OnTouchWake,
        .on_click        = &OnTouchClick,
        .on_double_click = &OnTouchDoubleClick,
    };
    if (axs5106l_touch_init(&cfg, &touch_) != ESP_OK) {
        ESP_LOGE(TAG, "touch init 失败");
        touch_ = nullptr;
    }
}

void InitializeTouchInput() {
    if (touch_) axs5106l_touch_attach_lvgl(touch_);
}

static void OnTouchWake(void *ctx) {
    static_cast<MyDazyP30_4GBoard*>(ctx)->WakeUp();
}

static void OnTouchClick(int16_t x, int16_t y, void *ctx) {
    auto* self = static_cast<MyDazyP30_4GBoard*>(ctx);
    self->WakeUp();
    if (y < 36) return;                          /* 状态栏交给 LVGL */
    self->HandleTouchSingleClick();
}

static void OnTouchDoubleClick(int16_t x, int16_t y, void *ctx) {
    auto* self = static_cast<MyDazyP30_4GBoard*>(ctx);
    self->WakeUp();
    if (y < 36) return;
    self->HandleTouchDoubleClick();
}

void EnterDeepSleep(...) {
    if (touch_) axs5106l_touch_sleep(touch_);
    ShutdownTouchAndAudioForSleep();
    /* esp_deep_sleep_start ... */
}

9.3 P31 · 加 swipe 控制中心

axs5106l_touch_config_t cfg = {
    /* ... 硬件 + rf_mode = RF_NORMAL ... */
    .on_wake  = &OnTouchWake,
    .on_click = &OnTouchClick,
    .on_swipe = &OnTouchSwipe,
};

static void OnTouchSwipe(int16_t dx, int16_t dy, void *ctx) {
    auto* self = static_cast<MyDazyP31Board*>(ctx);
    if (abs(dx) > abs(dy)) return;                /* 仅纵向 */
    auto* lcd = dynamic_cast<UiDisplay*>(self->GetDisplay());
    if (!lcd || lcd->IsBrainInfoVisible()) return;
    if (dy > 0) {
        if (!lcd->IsControlCenterVisible()) lcd->ShowControlCenter();
    } else {
        if (lcd->IsControlCenterVisible()) lcd->HideControlCenter();
        else if (lcd->IsMenuVisible())     lcd->HideMenu();
    }
}

11 注意事项

11.1 关键铁律

① 两阶段 init 顺序：GPIO → I²C → touch_init → SPI → LCD → LVGL → attach_lvgl。RST 与 LCD 共享 · 颠倒会把已点亮 LCD 一起 reset。
② sleep 时机：必须先于 AUDIO_PWR_EN=0 · 否则 codec 失电后 ESD 短路 SDA/SCL · I²C 写 sleep 失败（与 sc7a20h_wakeup 同时序铁律）。
③ ISR 约束：仅 ++int_edge_count · 不读 I²C · 不调阻塞 API · 不调 ESP_LOG。
